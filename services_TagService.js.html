import { PouchService } from './PouchService'

/**
 * Provides access to the tag database through a standardized interface.
 *
 * @class
 */
export class TagService {
    constructor () {
      this.pouchService = new PouchService('tags')
    }

    /**
     * Creates a new tag in the database.
     *
     * @param {string} name - The name of the tag.
     * @return {string} The ID of the newly created tag.
     *
     * @function
     */
    async create (name) {
      var document = {
        name: name
      }

      var id = this.pouchService.create(document)
      return id
    }

    /**
     * Gets the tag with the specified ID from the database.
     *
     * @param {string} tagId - The ID of the tag to get.
     * @return {Promise<Object>} - A promise to the requested tag object.
     *
     * @function
     */
    async getOne (tagId) {
      return this.pouchService.getOne(tagId)
    }

    /**
     * Gets all tags from the database.
     *
     * @return {Promise<Array<Object>>} - A promise to an array containing all tag objects.
     *
     * @function
     */
    async getAll () {
      return this.pouchService.getAll()
    }

    /**
     * Updates the name of the specified tag.
     *
     * @param {string} tagId - The ID of the tag to update the name for.
     * @param {string} name - The new name.
     *
     * @function
     */
    async updateName (tagId, name) {
      var document = await this.pouchService.getOne(tagId)
      document.name = name
      this.pouchService.update(document)
    }

    /**
     * Removes the specified tag from the database.
     *
     * @param {string} tagId - The ID of the tag to remove.
     *
     * @function
     */
    async delete (tagId) {
      var document = await this.pouchService.getOne(tagId)
      this.pouchService.delete(document)
    }
}
